IN = All packets coming IN to interface
OUT = All packets coming OUT to interface
NETWORK = 0.0.0.0/0 or X.X.X.X/X
ADDRESS = 0.0.0.0 or X.X.X.X
MASK = 255.255.255.255 or X.X.X.X

ACCESS-LIST = DENY AND PERMIT ONLY
ROUTE-MAP = DENY/PERMIT IN SPECIFIC ROUTE AND ROUTE IN ANOTHER 

FILTER-LIST/PREFIX-LIST/DISTRIBUTE-LIST/ROUTE-MAPS

ACCESS-GROUP/DISTRIBUTE-LIST = COMMAND THAT THEY HAVE IN AND OUT

CONFIGURATION FOR ACL STATIC ROUTE
	Router(config)#access-list 1 deny host ANY_HOST
	Router(config-if)#ip access-group 1 in

CONFIGURATION FOR ACL BGP ROUTE
	Router(config)#access-list 1 deny host ANY_HOST
	Router(config)#router bgp ANY_NUMBER
	Router(config-router)#distribute-list 1 in

CONFIGURATIONF FOR ROUTE MAP STATIC ROUTE
 	Router(config)#access-list 1 permit host ANY_HOST
	Router(config)#route-map ANY_NAME 
	Router(config-route-map)#match ip address 1
	Router(config-route-map)#set interface ANY_INTERFACE or Router(config-route-map)#set ip next-hop ANY_ADDRESS

	Router(config)#int ANY_INTERFACE
	Router(config-if)#ip policy route-map ANY_NAME

CONFIGURATION FOR STATIC ROUTE
	Router(config)#ip route 0.0.0.0 0.0.0.0 ANY_ADDRESS -> DEFAULT ROUTE
	Router(config)#ip route ANY_NETWORK ANY_MASK ANY_ADDRESS

CONFIGURATION FOR IP 
	Router(config)#int ANY_INTEFACE
	Router(config-if)#ip address ANY_ADDRESS ANY_MASK
	Router(config-if)#no shutdown
